<section id="rsvp" class="fundo-cinza">
        <h2>Confirme sua Presença</h2>
        <p style="color: var(--texto-secundario); margin-bottom: 30px;">
            Por favor, confirme sua presença até o dia 15 de Julho de 2026.
        </p>

        <form id="form-rsvp" style="max-width: 400px; margin: 0 auto; display: flex; flex-direction: column; gap: 15px;">
            <input type="text" id="nomeConvidado" placeholder="Seu Nome Completo" required 
                   style="padding: 15px; border: 1px solid #ddd; border-radius: 5px; font-family: var(--fonte-textos); font-size: 1rem;">
            
            <input type="number" id="qtdAcompanhantes" placeholder="Número de Acompanhantes (ex: 2)" min="0" required 
                   style="padding: 15px; border: 1px solid #ddd; border-radius: 5px; font-family: var(--fonte-textos); font-size: 1rem;">
            
            <button type="submit" id="btn-confirmar"
                    style="padding: 15px; background-color: var(--texto-principal); color: var(--branco-puro); border: none; border-radius: 5px; font-size: 1rem; cursor: pointer; text-transform: uppercase; letter-spacing: 1px; transition: background-color 0.3s;">
                Confirmar Presença
            </button>
            
            <p id="mensagem-feedback" style="display: none; font-weight: bold; margin-top: 10px;"></p>
        </form>
    </section>

    <script type="module">
        // Importa as funções necessárias do Firebase
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.1/firebase-app.js";
        import { getFirestore, collection, addDoc } from "https://www.gstatic.com/firebasejs/10.8.1/firebase-firestore.js";

        // COLOQUE AQUI AS SUAS CHAVES DO FIREBASE (Passo 1)
        const firebaseConfig = {
            apiKey: "SUA_API_KEY",
            authDomain: "seu-projeto.firebaseapp.com",
            projectId: "seu-projeto",
            storageBucket: "seu-projeto.appspot.com",
            messagingSenderId: "123456789",
            appId: "SEU_APP_ID"
        };

        // Inicializa o Firebase e o Firestore
        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);

        // Lógica de Envio do Formulário
        document.getElementById('form-rsvp').addEventListener('submit', async function(event) {
            event.preventDefault(); // Evita recarregar a página

            const btn = document.getElementById('btn-confirmar');
            const feedback = document.getElementById('mensagem-feedback');
            
            // Pega os valores
            const nome = document.getElementById('nomeConvidado').value;
            // Converte o valor para número inteiro para podermos somar no banco de dados
            const acompanhantes = parseInt(document.getElementById('qtdAcompanhantes').value, 10);

            try {
                // Muda o texto do botão para indicar carregamento
                btn.innerText = "Enviando...";
                btn.disabled = true;

                // Salva no banco de dados na coleção "convidados"
                const docRef = await addDoc(collection(db, "convidados"), {
                    nome: nome,
                    acompanhantes: acompanhantes,
                    dataConfirmacao: new Date().toISOString() // Salva o dia e hora exatos
                });

                // Sucesso
                feedback.style.display = "block";
                feedback.style.color = "green";
                feedback.innerText = "Presença confirmada com sucesso! Muito obrigado.";
                
                // Limpa o formulário
                document.getElementById('form-rsvp').reset();

            } catch (e) {
                // Erro
                console.error("Erro ao adicionar documento: ", e);
                feedback.style.display = "block";
                feedback.style.color = "red";
                feedback.innerText = "Ocorreu um erro ao confirmar. Tente novamente.";
            } finally {
                // Volta o botão ao normal
                btn.innerText = "Confirmar Presença";
                btn.disabled = false;
            }
        });
    </script>
