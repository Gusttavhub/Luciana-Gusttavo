    <script>
        // Lógica da Contagem Regressiva
        // Define a data alvo: 15 de Agosto de 2026 às 18:00:00
        const dataCasamento = new Date("August 15, 2026 18:00:00").getTime();

        const atualizarContador = setInterval(function() {
            const agora = new Date().getTime();
            const distancia = dataCasamento - agora;

            // Cálculos matemáticos para dias, horas, minutos e segundos
            const dias = Math.floor(distancia / (1000 * 60 * 60 * 24));
            const horas = Math.floor((distancia % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutos = Math.floor((distancia % (1000 * 60 * 60)) / (1000 * 60));
            const segundos = Math.floor((distancia % (1000 * 60)) / 1000);

            // Injeta o HTML no elemento do contador
            document.getElementById("contador").innerHTML = `
                <div class="bloco-tempo">
                    <span class="numero">${dias}</span>
                    <span class="legenda">Dias</span>
                </div>
                <div class="bloco-tempo">
                    <span class="numero">${horas.toString().padStart(2, '0')}</span>
                    <span class="legenda">Horas</span>
                </div>
                <div class="bloco-tempo">
                    <span class="numero">${minutos.toString().padStart(2, '0')}</span>
                    <span class="legenda">Min</span>
                </div>
                <div class="bloco-tempo">
                    <span class="numero">${segundos.toString().padStart(2, '0')}</span>
                    <span class="legenda">Seg</span>
                </div>
            `;

            // Se a data já passou, limpa o intervalo e exibe uma mensagem
            if (distancia < 0) {
                clearInterval(atualizarContador);
                document.getElementById("contador").innerHTML = "<p class='numero'>Chegou o grande dia!</p>";
            }
        }, 1000);
    </script>
